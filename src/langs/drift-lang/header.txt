Bit0:
  bit0 = Bit0

Bit1:
  bit1 = Bit1

Bit:
  bit a:Bit0 = Bit a
  bit a:Bit1 = Bit a

----------------------------------------------------------------------------------------------------

Nil:
  nil ~t = Nil ~t

Cons:
  cons a:(t *) (List ~t b) = Cons ~t a b

List:
  list a:(Nil ~t) = List ~t a
  list a:(Cons ~t * *) = List ~t a
